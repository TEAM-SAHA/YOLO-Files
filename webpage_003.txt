<script>
  /* ── State ── */
  const state = {
    connected: true,
    battPct:   78,
    signalBars:3,
    crackCount:3,
    uptime:    15727, // seconds
    fps:       24,
    detecting: false,
  };

  /* ── Helpers ── */
  function pad(n){ return String(n).padStart(2,'0'); }

  function nowStr(){
    const d=new Date();
    return `${pad(d.getHours())}:${pad(d.getMinutes())}:${pad(d.getSeconds())}`;
  }

  function fmtUptime(s){
    const h=Math.floor(s/3600), m=Math.floor((s%3600)/60), sec=s%60;
    return `${pad(h)}:${pad(m)}:${pad(sec)}`;
  }

  /* ── Toast ── */
  let toastTimer;
  function showToast(msg){
    const t=document.getElementById('toast');
    t.textContent=msg;
    t.classList.add('show');
    clearTimeout(toastTimer);
    toastTimer=setTimeout(()=>t.classList.remove('show'),2200);
  }

  /* ── Log ── */
  const logTypes = {
    'FORWARD':  { cls:'info',  text:'Forward command sent' },
    'BACK':     { cls:'info',  text:'Backward command sent' },
    'LEFT':     { cls:'info',  text:'Left turn command sent' },
    'RIGHT':    { cls:'info',  text:'Right turn command sent' },
    'STOP':     { cls:'stop',  text:'Robot stopped' },
    'CRACK':    { cls:'crack', text:'⚠ Crack detected — confidence 87.4%' },
    'CLEAR':    { cls:'ok',    text:'✓ Area cleared — no cracks detected' },
    'ESTOP':    { cls:'stop',  text:'🛑 EMERGENCY STOP activated' },
    'CONNECT':  { cls:'ok',    text:'✓ Connection established' },
    'DISCONN':  { cls:'stop',  text:'⚠ Connection lost' },
    'BOOT':     { cls:'info',  text:'System boot complete — all modules online' },
    'SERVO':    { cls:'info',  text:'Servo calibration OK (8/8)' },
    'CAM':      { cls:'info',  text:'Camera stream initialized — 640×480 @ 24fps' },
  };

  function addLog(type, customText){
    const console = document.getElementById('logConsole');
    const entry   = logTypes[type] || { cls:'info', text: customText||type };
    const div     = document.createElement('div');
    div.className = `log-entry ${entry.cls}`;
    div.innerHTML = `<span class="log-time">[${nowStr()}]</span><span class="log-text"> ${customText||entry.text}</span>`;
    console.appendChild(div);
    console.scrollTop = console.scrollHeight;
    // keep max 120 lines
    while(console.children.length > 120) console.removeChild(console.firstChild);
  }

  function clearLog(){
    document.getElementById('logConsole').innerHTML='';
    addLog('BOOT');
    showToast('Log cleared');
  }

  /* ── Commands ── */
  function sendCmd(cmd){
    if(!state.connected){ showToast('Not connected'); return; }
    addLog(cmd);
    showToast(`${cmd} →`);
    // Here you would: fetch(`http://192.168.1.100/${cmd.toLowerCase()}`).catch(console.warn);
  }

  /* ── Emergency Stop ── */
  function emergencyStop(){
    sendCmd('STOP');
    addLog('ESTOP');
    showToast('🛑 EMERGENCY STOP');
    document.getElementById('sMode').textContent = 'HALTED';
    document.getElementById('sMode').className  = 'sc-val warn';
  }

  /* ── Connection toggle (demo) ── */
  function toggleConnection(){
    state.connected = !state.connected;
    const dot  = document.getElementById('connDot');
    const txt  = document.getElementById('connText');
    if(state.connected){
      dot.classList.remove('disconnected');
      txt.textContent = 'CONNECTED';
      txt.style.color = 'var(--green)';
      addLog('CONNECT');
    } else {
      dot.classList.add('disconnected');
      txt.textContent = 'DISCONNECTED';
      txt.style.color = 'var(--red)';
      addLog('DISCONN');
    }
  }

  /* ── Sliders ── */
  function updateSlider(id, val, unit){
    const map = { amp:'ampVal', speed:'speedVal', bright:'brightVal' };
    const el  = document.getElementById(map[id]);
    const disp= id==='speed' ? parseFloat(val).toFixed(2) : Math.round(val);
    el.textContent = disp + unit;
    if(id==='speed'){
      document.getElementById('sSpeed').textContent  = parseFloat(val).toFixed(2)+' Hz';
    }
  }

  /* ── Clock / HUD time ── */
  function updateClock(){
    const t=nowStr();
    document.getElementById('hudTime').textContent = t;
  }

  /* ── Uptime ── */
  function updateUptime(){
    state.uptime++;
    const s=fmtUptime(state.uptime);
    document.getElementById('sUptime').textContent = s;
    document.getElementById('uptime').textContent  = 'UPTIME\u00A0'+s;
  }

  /* ── Simulated telemetry ── */
  let tickCount = 0;
  function telemetryTick(){
    tickCount++;

    /* battery drift */
    if(tickCount % 40 === 0 && state.battPct > 5){
      state.battPct = Math.max(5, state.battPct - 0.5);
      const pct = state.battPct;
      document.getElementById('battFill').style.width = pct+'%';
      document.getElementById('battVal').textContent  = Math.round(pct)+'%';
      document.getElementById('battFill').style.background =
        pct < 20 ? 'linear-gradient(90deg,var(--red),#ff6060aa)'
        : pct < 40 ? 'linear-gradient(90deg,var(--yellow),#ffd740aa)'
        : 'linear-gradient(90deg,var(--green),#00e676aa)';
    }

    /* FPS jitter */
    const fps = 22 + Math.floor(Math.random()*5);
    document.getElementById('hudFPS').textContent  = `FPS: ${fps}`;
    document.getElementById('sFPS').textContent    = `${fps} fps`;

    /* latency jitter */
    const lat = 20 + Math.floor(Math.random()*30);
    document.getElementById('latency').textContent = `${lat}ms`;

    /* CPU temp drift */
    const temp = 44 + Math.floor(Math.random()*8);
    const tEl  = document.getElementById('sTemp');
    tEl.textContent = `${temp} °C`;
    tEl.className   = `sc-val ${temp > 55 ? 'warn' : ''}`;

    /* signal drift */
    if(tickCount % 25 === 0){
      state.signalBars = Math.max(1, Math.min(4, state.signalBars + (Math.random()>.6 ? 1 : -1)));
      const bars = document.querySelectorAll('#signalBars span');
      bars.forEach((b,i)=> b.classList.toggle('active', i < state.signalBars));
      const dbm = -45 - (4-state.signalBars)*15;
      document.getElementById('sigLabel').textContent = `${dbm} dBm`;
    }

    /* simulate crack detection event roughly every 90s */
    if(tickCount % 180 === 90 && state.connected){
      triggerCrackDetection();
    }
    if(tickCount % 180 === 140 && state.connected){
      clearCrackDetection();
    }
  }

  function triggerCrackDetection(){
    state.crackCount++;
    state.detecting = true;
    document.getElementById('detectBox').classList.add('active');
    document.getElementById('detStatus').textContent = 'DETECTED';
    document.getElementById('detStatus').className   = 'cc-val danger';
    document.getElementById('detStatusCard').classList.add('alert');
    document.getElementById('lastDetect').textContent = nowStr();
    document.getElementById('crackCount').textContent = state.crackCount;
    document.getElementById('sDetect').textContent = 'DETECTED';
    document.getElementById('sDetect').className   = 'sc-val danger';
    const conf = (82 + Math.random()*15).toFixed(1);
    document.getElementById('detConf').textContent = conf+'%';
    document.getElementById('confFill').style.width = conf+'%';
    addLog('CRACK');
  }

  function clearCrackDetection(){
    state.detecting = false;
    document.getElementById('detectBox').classList.remove('active');
    document.getElementById('detStatus').textContent = 'CLEAR';
    document.getElementById('detStatus').className   = 'cc-val ok';
    document.getElementById('detStatusCard').classList.remove('alert');
    document.getElementById('sDetect').textContent = 'SCANNING';
    document.getElementById('sDetect').className   = 'sc-val warn';
    addLog('CLEAR');
  }

  /* ── Boot sequence ── */
  function boot(){
    addLog('BOOT');
    setTimeout(()=>addLog('SERVO'), 600);
    setTimeout(()=>addLog('CAM'),   1200);
    setTimeout(()=>addLog('CONNECT'),1900);

    // seed a few historical entries
    setTimeout(()=>{
      addLog('FORWARD');
      setTimeout(()=>{ addLog('CRACK'); triggerCrackDetection(); }, 600);
      setTimeout(()=>{ addLog('STOP'); clearCrackDetection(); }, 1200);
    }, 2400);
  }

  /* ── Start ── */
  boot();
  setInterval(updateClock, 1000);
  setInterval(updateUptime, 1000);
  setInterval(telemetryTick, 500);
</script>
